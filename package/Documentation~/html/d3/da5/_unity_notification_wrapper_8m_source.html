87 void _ScheduleLocalNotification(struct iOSNotificationData* data)
88 {
89  UnityNotificationManager* manager = [UnityNotificationManager sharedInstance];
90  [manager requestAuthorization:(UNAuthorizationOptionSound + UNAuthorizationOptionAlert + UNAuthorizationOptionBadge) : YES];
91 
92  assert(manager.onNotificationReceivedCallback != NULL);
93 
94  NSDictionary *userInfo = @{
95  @"showInForeground" : @(data->showInForeground),
96  @"showInForegroundPresentationOptions" : [NSNumber numberWithInteger:data->showInForegroundPresentationOptions]
97  };
98 
99  UNMutableNotificationContent* content = [[UNMutableNotificationContent alloc] init];
100  content.title = [NSString localizedUserNotificationStringForKey: [NSString stringWithUTF8String: data->title] arguments:nil];
101  content.body = [NSString localizedUserNotificationStringForKey: [NSString stringWithUTF8String: data->body] arguments:nil];
102  content.badge = [NSNumber numberWithInt:data->badge];
103  content.userInfo = userInfo;
104 
105  if (data->subtitle != NULL)
106  content.subtitle = [NSString localizedUserNotificationStringForKey: [NSString stringWithUTF8String: data->subtitle] arguments:nil];
107 
108  if (data->categoryIdentifier != NULL)
109  content.categoryIdentifier = [NSString stringWithUTF8String:data->categoryIdentifier];
110 
111  if (data->threadIdentifier != NULL)
112  content.threadIdentifier = [NSString stringWithUTF8String:data->threadIdentifier];
113 
114  // TODO add a way to specify custom sounds.
115  content.sound = [UNNotificationSound defaultSound];
116 
117  UNNotificationTrigger* trigger;
118 
119  if ( data->triggerType == 0)
120  {
121  trigger = [UNTimeIntervalNotificationTrigger triggerWithTimeInterval:data->timeTriggerInterval repeats: data -> repeats];
122  }
123  else if ( data->triggerType == 1)
124  {
125  NSDateComponents* date = [[NSDateComponents alloc] init];
126  if ( data->calendarTriggerYear >= 0)
127  date.year = data->calendarTriggerYear;
128  if (data->calendarTriggerMonth >= 0)
129  date.hour = data->calendarTriggerMonth;
130  if (data->calendarTriggerDay >= 0)
131  date.hour = data->calendarTriggerDay;
132  if (data->calendarTriggerHour >= 0)
133  date.hour = data->calendarTriggerHour;
134  if (data->calendarTriggerMinute >= 0)
135  date.hour = data->calendarTriggerMinute;
136  if (data->calendarTriggerSecond >= 0)
137  date.hour = data->calendarTriggerSecond;
138 
139  trigger = [UNCalendarNotificationTrigger triggerWithDateMatchingComponents:date repeats:data->repeats];
140  }
141  else if ( data->triggerType == 2)
142  {
143 #ifdef defined(UNITY_USES_LOCATION) && UNITY_USES_LOCATION
144  CLLocationCoordinate2D center = CLLocationCoordinate2DMake(data->locationTriggerCenterX, data->locationTriggerCenterY);
145 
146  CLCircularRegion* region = [[CLCircularRegion alloc] initWithCenter:center
147  radius:data->locationTriggerRadius identifier:@"Headquarters"];
148  region.notifyOnEntry = data->locationTriggerNotifyOnEntry;
149  region.notifyOnExit = data->locationTriggerNotifyOnExit;
150 
151  trigger = [UNLocationNotificationTrigger triggerWithRegion:region repeats:NO];
152 #endif
153  }
154  else
155  {
156  //Should throw an error TODO to managed.
157  return;
158  }
159 
160  UNNotificationRequest* request = [UNNotificationRequest requestWithIdentifier:
161  [NSString stringWithUTF8String:data->identifier] content:content trigger:trigger];
162 
163  // Schedule the notification.
164  UNUserNotificationCenter* center = [UNUserNotificationCenter currentNotificationCenter];
165  [center addNotificationRequest:request withCompletionHandler:^(NSError * _Nullable error) {
166 
167  if (error != NULL)
168  NSLog(@"%@",[error localizedDescription]);
169 
170  [manager updateScheduledNotificationList];
171 
172  }];
173 }
