68  authData -> finished = YES;

70  authData -> error = [[error localizedDescription]cStringUsingEncoding:NSUTF8StringEncoding];
71  authData -> deviceToken = "";
72 
73  self.authData = authData;
74 
75  if (granted)
76  {
77  self.authorized = TRUE;
78  if (registerRemote)

80  dispatch_async(dispatch_get_main_queue(), ^{

123 
124  if (self.onNotificationReceivedCallback != NULL )
125  self.onNotificationReceivedCallback([UnityNotificationManager UNNotificationToiOSNotificationData:notification]);

128 // UNNotificationPresentationOptionAlert | UNNotificationPresentationOptionSound | UNNotificationPresentationOptionBadge

130 // int a = UNNotificationPresentationOptionAlert;
131 // int b = UNNotificationPresentationOptionSound;
132 // int c = UNNotificationPresentationOptionBadge;
133 // int s = (int)UNNotificationPresentationOptionNone;
134 
135  BOOL showInForeground;
136  NSInteger presentationOptions;
137 
138  if ( [notification.request.trigger isKindOfClass:[UNPushNotificationTrigger class]])
139  {

154  showInForeground = [[notification.request.content.userInfo objectForKey:@"showInForeground"] boolValue];
155 
156  }
157  if (showInForeground)
158  {
159  completionHandler(presentationOptions);
160  }
161  else
162  completionHandler(UNNotificationPresentationOptionNone);
163 
164  [[UnityNotificationManager sharedInstance] updateDeliveredNotificationList];

215 {
216  struct iOSNotificationData* notificationData = (struct iOSNotificationData*)malloc(sizeof(*notificationData));

218  UNNotificationContent* content = request.content;

228  notificationData -> body = [content.body UTF8String];

232  notificationData -> badge = content.badge;
233 
234  if (content.subtitle != nil && content.subtitle.length > 0)
235  notificationData -> subtitle = [content.subtitle UTF8String];
236  else
237  notificationData -> subtitle = " ";
238 
239  if (content.categoryIdentifier != nil && content.categoryIdentifier.length > 0)
240  notificationData -> categoryIdentifier = [content.categoryIdentifier UTF8String];
241  else
242  notificationData -> categoryIdentifier = " ";
243 
244  if (content.threadIdentifier != nil && content.threadIdentifier.length > 0)
245  notificationData -> threadIdentifier = [content.threadIdentifier UTF8String];
246  else
247  notificationData -> threadIdentifier = " ";
248 
249  //0 - time, 1 - calendar, 2 - location, 3 - push.
250  // TODO initialize proper values!
251  if ([ request.trigger isKindOfClass:[UNTimeIntervalNotificationTrigger class]])
252  {
253  notificationData -> triggerType = TIME_TRIGGER;
254  notificationData -> timeTriggerInterval = 0;
255  notificationData -> repeats = request.trigger.repeats;
256  }
257  else if ([ request.trigger isKindOfClass:[UNCalendarNotificationTrigger class]])
258  {
259  UNLocationNotificationTrigger* locTrigger = (UNLocationNotificationTrigger*) request.trigger;
260  notificationData -> triggerType = CALENDAR_TRIGGER;
261  notificationData -> locationTriggerCenterX = 0;
262  notificationData -> locationTriggerCenterY = 0;
263  notificationData -> locationTriggerRadius = 0;
264  notificationData -> locationTriggerNotifyOnExit = 0;
265  notificationData -> locationTriggerNotifyOnEntry = 0;
266  }
267  else if ([ request.trigger isKindOfClass:[UNLocationNotificationTrigger class]])
268  {
269  notificationData -> triggerType = LOCATION_TRIGGER;
270  }
271  else if ([ request.trigger isKindOfClass:[UNPushNotificationTrigger class]])
272  {
273  notificationData -> triggerType = PUSH_TRIGGER;
274  }
275 
276  return notificationData;
