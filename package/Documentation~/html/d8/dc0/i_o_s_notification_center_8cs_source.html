327  public iOSNotificationTrigger Trigger
328  {
329  set
330  {
331  if (value is iOSNotificationTimeIntervalTrigger)
332  {
333  var trigger = (iOSNotificationTimeIntervalTrigger) value;
334  data.triggerType = iOSNotificationTimeIntervalTrigger.Type;
335  data.timeTriggerInterval = trigger.timeInterval;
336 
337  if (trigger.timeInterval > 60)
338  {
339  data.repeats = trigger.Repeats;
340  }
341  else
342  {
343  if (trigger.Repeats)
344  {
345  Debug.LogWarning("Time interval must be at least 60 for repeating notifications.");
346  }
347  }
348 
349  }
350  else if (value is iOSNotificationCalendarTrigger)
351  {
352  var trigger = (iOSNotificationCalendarTrigger) value;
353  data.triggerType = iOSNotificationCalendarTrigger.Type;
354  data.calendarTriggerYear = trigger.Year != null ? trigger.Year.Value : -1;
355  data.calendarTriggerMonth = trigger.Month != null ? trigger.Month.Value : -1;
356  data.calendarTriggerDay = trigger.Day != null ? trigger.Day.Value : -1;
357  data.calendarTriggerHour = trigger.Hour != null ? trigger.Hour.Value : -1;
358  data.calendarTriggerMinute = trigger.Minute != null ? trigger.Minute.Value : -1;
359  data.calendarTriggerSecond = trigger.Second != null ? trigger.Second.Value : -1;
360  data.repeats = trigger.Repeats;
361 
362  }
363  else if (value is iOSNotificationLocationTrigger)
364  {
365  var trigger = (iOSNotificationLocationTrigger) value;
366  data.triggerType = iOSNotificationLocationTrigger.Type;
367  data.locationTriggerCenterX = trigger.Center.x;
368  data.locationTriggerCenterY = trigger.Center.x;
369  data.locationTriggerNotifyOnEntry = trigger.NotifyOnEntry;
370  data.locationTriggerNotifyOnExit = trigger.NotifyOnExit;
371  data.locationTriggerRadius = trigger.Radius;
372  }
373  else if (value is iOSNotificationPushTrigger)
374  {
375  data.triggerType = 3;
376  }
377  }
378 
379  get
380  {
381  iOSNotificationTrigger trigger;
382  if (data.triggerType == iOSNotificationTimeIntervalTrigger.Type)
383  {
384  trigger = new iOSNotificationTimeIntervalTrigger()
385  {
386  timeInterval = data.timeTriggerInterval,
387  Repeats = data.repeats
388  };
389  }
390  else if (data.triggerType == iOSNotificationCalendarTrigger.Type)
391  {
392  trigger = new iOSNotificationCalendarTrigger()
393  {
394  Year = (data.calendarTriggerYear > 0) ? (int?) data.calendarTriggerYear : null,
395  Month = (data.calendarTriggerMonth > 0) ? (int?) data.calendarTriggerMonth : null,
396  Day = (data.calendarTriggerDay > 0) ? (int?) data.calendarTriggerDay : null,
397  Hour = (data.calendarTriggerHour >= 0) ? (int?) data.calendarTriggerHour : null,
398  Minute = (data.calendarTriggerMinute >= 0) ? (int?) data.calendarTriggerMinute : null,
399  Second = (data.calendarTriggerSecond >= 0) ? (int?) data.calendarTriggerSecond : null,
400  Repeats = data.repeats
401  };
402  }
403  else if (data.triggerType == iOSNotificationLocationTrigger.Type)
404  {
405  trigger = new iOSNotificationLocationTrigger()
406  {
407  Center = new Vector2(data.locationTriggerCenterX, data.locationTriggerCenterY),
408  Radius = data.locationTriggerRadius,
409  NotifyOnEntry = data.locationTriggerNotifyOnEntry,
410  NotifyOnExit = data.locationTriggerNotifyOnExit
411  };
412  }
413  else
414  {
415  trigger = new iOSNotificationPushTrigger();
416  }
417  return trigger;
418  }
