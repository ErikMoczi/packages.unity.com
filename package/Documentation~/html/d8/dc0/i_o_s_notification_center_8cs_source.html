353  public iOSNotificationTrigger Trigger
354  {
355  set
356  {
357  if (value is iOSNotificationTimeIntervalTrigger)
358  {
359  var trigger = (iOSNotificationTimeIntervalTrigger) value;
360  data.triggerType = iOSNotificationTimeIntervalTrigger.Type;
361  data.timeTriggerInterval = trigger.timeInterval;
362 
363  if (trigger.timeInterval > 60)
364  {
365  data.repeats = trigger.Repeats;
366  }
367  else
368  {
369  if (trigger.Repeats)
370  {
371  Debug.LogWarning("Time interval must be at least 60 for repeating notifications.");
372  }
373  }
374 
375  }
376  else if (value is iOSNotificationCalendarTrigger)
377  {
378  var trigger = (iOSNotificationCalendarTrigger) value;
379  data.triggerType = iOSNotificationCalendarTrigger.Type;
380  data.calendarTriggerYear = trigger.Year != null ? trigger.Year.Value : -1;
381  data.calendarTriggerMonth = trigger.Month != null ? trigger.Month.Value : -1;
382  data.calendarTriggerDay = trigger.Day != null ? trigger.Day.Value : -1;
383  data.calendarTriggerHour = trigger.Hour != null ? trigger.Hour.Value : -1;
384  data.calendarTriggerMinute = trigger.Minute != null ? trigger.Minute.Value : -1;
385  data.calendarTriggerSecond = trigger.Second != null ? trigger.Second.Value : -1;
386  data.repeats = trigger.Repeats;
387 
388  }
389  else if (value is iOSNotificationLocationTrigger)
390  {
391  var trigger = (iOSNotificationLocationTrigger) value;
392  data.triggerType = iOSNotificationLocationTrigger.Type;
393  data.locationTriggerCenterX = trigger.Center.x;
394  data.locationTriggerCenterY = trigger.Center.x;
395  data.locationTriggerNotifyOnEntry = trigger.NotifyOnEntry;
396  data.locationTriggerNotifyOnExit = trigger.NotifyOnExit;
397  data.locationTriggerRadius = trigger.Radius;
398  }
399  else if (value is iOSNotificationPushTrigger)
400  {
401  data.triggerType = 3;
402  }
403  }
404 
405  get
406  {
407  iOSNotificationTrigger trigger;
408  if (data.triggerType == iOSNotificationTimeIntervalTrigger.Type)
409  {
410  trigger = new iOSNotificationTimeIntervalTrigger()
411  {
412  timeInterval = data.timeTriggerInterval,
413  Repeats = data.repeats
414  };
415  }
416  else if (data.triggerType == iOSNotificationCalendarTrigger.Type)
417  {
418  trigger = new iOSNotificationCalendarTrigger()
419  {
420  Year = (data.calendarTriggerYear > 0) ? (int?) data.calendarTriggerYear : null,
421  Month = (data.calendarTriggerMonth > 0) ? (int?) data.calendarTriggerMonth : null,
422  Day = (data.calendarTriggerDay > 0) ? (int?) data.calendarTriggerDay : null,
423  Hour = (data.calendarTriggerHour >= 0) ? (int?) data.calendarTriggerHour : null,
424  Minute = (data.calendarTriggerMinute >= 0) ? (int?) data.calendarTriggerMinute : null,
425  Second = (data.calendarTriggerSecond >= 0) ? (int?) data.calendarTriggerSecond : null,
426  Repeats = data.repeats
427  };
428  }
429  else if (data.triggerType == iOSNotificationLocationTrigger.Type)
430  {
431  trigger = new iOSNotificationLocationTrigger()
432  {
433  Center = new Vector2(data.locationTriggerCenterX, data.locationTriggerCenterY),
434  Radius = data.locationTriggerRadius,
435  NotifyOnEntry = data.locationTriggerNotifyOnEntry,
436  NotifyOnExit = data.locationTriggerNotifyOnExit
437  };
438  }
439  else
440  {
441  trigger = new iOSNotificationPushTrigger();
442  }
443  return trigger;
444  }
